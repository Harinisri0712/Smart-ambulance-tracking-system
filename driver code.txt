    <script>

            authDomain: "ambulance-tracking-9b928.firebaseapp.com",
            databaseURL: "https://ambulance-tracking-9b928-default-rtdb.asia-southeast1.firebasedatabase.app",
            projectId: "ambulance-tracking-9b928",
            storageBucket: "ambulance-tracking-9b928.firebasestorage.app",
            messagingSenderId: "609779802352",
            appId: "1:609779802352:web:4f9eefa261936311c12235"
        };

        firebase.initializeApp(firebaseConfig);
        const db = firebase.database();

        const nachipatty = [11.513973, 78.077053]; 
        const vennandur = [11.52, 78.09];

        const map = L.map('map').setView(nachipatty, 14);
        L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png').addTo(map);

        const driverMarker = L.marker(nachipatty).addTo(map).bindPopup("Ambulance");
        const redIcon = L.icon({
            iconUrl: 'https://raw.githubusercontent.com/pointhi/leaflet-color-markers/master/img/marker-icon-2x-red.png',
            iconSize: [25, 41], iconAnchor: [12, 41]
        });
        L.marker(vennandur, {icon: redIcon}).addTo(map).bindPopup("Victim Site");

        let roadPoints = [];

        const control = L.Routing.control({
            waypoints: [L.latLng(nachipatty), L.latLng(vennandur)],
            lineOptions: { styles: [{ color: '#1976d2', weight: 6 }] },
            show: false,
            addWaypoints: false
        }).addTo(map);

        control.on('routesfound', function(e) {
            roadPoints = e.routes[0].coordinates;
            const summary = e.routes[0].summary;
            
            document.getElementById('route-info').innerHTML = 
                `Distance: ${(summary.totalDistance/1000).toFixed(2)} km<br>` +
                `Est. Time: ${Math.round(summary.totalTime/60)} mins`;
            
            document.getElementById('startBtn').innerText = "START MISSION 🚑";
            document.getElementById('startBtn').disabled = false;
        });

        function animateAlongRoad() {
            let i = 0;
            const interval = setInterval(() => {
                if (i >= roadPoints.length) {
                    clearInterval(interval);
                    document.getElementById('startBtn').innerText = "ARRIVED ✅";
                    return;
                }

                const currentPoint = roadPoints[i];
                driverMarker.setLatLng([currentPoint.lat, currentPoint.lng]);
                
                db.ref('Ambulance').set({ 
                    lat: currentPoint.lat, 
                    lng: currentPoint.lng,
                    timestamp: Date.now(),
                    status: "Moving to Accident Site"
                });

                i += 5;
            }, 1000);
        }

        document.getElementById('startBtn').addEventListener('click', () => {
            document.getElementById('startBtn').disabled = true;
            document.getElementById('startBtn').innerText = "EN ROUTE...";
            animateAlongRoad();
        });
